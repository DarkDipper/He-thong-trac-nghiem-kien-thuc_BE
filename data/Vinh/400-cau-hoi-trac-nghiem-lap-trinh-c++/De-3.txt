#include &lt;stdio.h&gt;
#include &lt;conio.h&gt;
 
int reverse(int);
 
void main()
{
    int n = 5;
    reverse(n);
    getch();
}
 
int reverse(int n)
{
    if(n == 0)
        return 0;
    else
        printf("%d", n);
    reverse(n--);
}
